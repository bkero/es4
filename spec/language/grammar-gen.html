= Grammar = == Name Expressions ==

Identifier   →
      Identifier
   | ContextuallyReservedIdentifier

PropertyIdentifier   →
      Identifier
   | ReservedIdentifier

NameExpression   →
      Identifier
   | NamespaceExpression :: PropertyIdentifier

 == Expressions == α = { allowColon, noColon }
β = { allowIn, noIn }

ParenExpression   →
      ( CommaExpressionallowColon, allowIn )

FunctionExpressionα, β   →
      function PropertyIdentifier FunctionSignature FunctionExpressionBodyα, β
   | function FunctionSignature FunctionExpressionBodyα, β

FunctionExpressionBodyα, β   →
      { Directiveslocal }
   | CommaExpressionα, β

ObjectInitialisernoColon   →
      InitialiserAttribute { FieldList }

ObjectInitialiserallowColon   →
      InitialiserAttribute { FieldList }
   | InitialiserAttribute { FieldList } : TypeExpression

FieldList   →
      «empty»
   | Field
   | Field , FieldList

Field   →
      InitialiserAttribute FieldName : AssignmentExpressionallowColon, allowIn
   | InitialiserAttribute get FieldName GetterSignature FunctionExpressionBodyallowColon, allowIn
   | InitialiserAttribute set FieldName SetterSignature FunctionExpressionBodyallowColon, allowIn
   | __proto__ : AssignmentExpressionallowColon, allowIn

InitialiserAttribute   →
      «empty»
   | const
   | var

FieldName   →
      NameExpression
   | StringLiteral
   | NumberLiteral
   | [lookahead !{__proto__}] ReservedIdentifier

ArrayInitialisernoColon   →
      InitialiserAttribute [ ArrayElements ]

ArrayInitialiserallowColon   →
      InitialiserAttribute [ ArrayElements ]
   | InitialiserAttribute [ ArrayElements ] : TypeExpression

ArrayElements   →
      ArrayElementList
   | ArrayComprehension

ArrayElementList   →
      «empty»
   | AssignmentExpressionallowColon, allowIn
   | SpreadExpression
   | , ArrayElementList
   | AssignmentExpressionallowColon, allowIn , ArrayElementList

SpreadExpression   →
      ... AssignmentExpressionallowColon, allowIn

ArrayComprehension   →
      AssignmentExpressionallowColon, allowIn ComprehensionExpression

ComprehensionExpression   →
      for ( TypedPatternnoIn in CommaExpressionallowColon, allowIn ) ComprehensionClause
   | for each ( TypedPatternnoIn in CommaExpressionallowColon, allowIn ) ComprehensionClause
   | let ( LetBindingList ) ComprehensionClause
   | if ParenExpression ComprehensionClause

ComprehensionClause   →
      «empty»
   | ComprehensionExpression

PrimaryExpressionα, β   →
      null
   | true
   | false
   | DoubleLiteral
   | DecimalLiteral
   | StringLiteral
   | RegExpInitialiser
   | ArrayInitialiserα
   | ObjectInitialiserα
   | FunctionExpressionα, β
   | ThisExpression
   | LetExpressionα, β
   | ParenExpression
   | NameExpression

ThisExpression   →
      this
   | this [no line break] function
   | this [no line break] generator

LetExpressionα, β   →
      let ( LetBindingList ) CommaExpressionα, β

LetBindingList   →
      «empty»
   | VariableBindingListallowIn

Arguments   →
      ( )
   | ( SpreadExpression )
   | ( ArgumentList )
   | ( ArgumentList , SpreadExpression )

ArgumentList   →
      AssignmentExpressionallowColon, allowIn
   | ArgumentList , AssignmentExpressionallowColon, allowIn

PropertyOperator   →
      . ReservedIdentifier
   | . NameExpression
   | Brackets
   | TypeApplication

Brackets   →
      [ CommaExpressionnoColon, allowIn ]
   | [ SliceExpression ]

SliceExpression   →
      OptionalExpressionnoColon : OptionalExpressionnoColon
   | OptionalExpressionnoColon : OptionalExpressionnoColon : OptionalExpressionallowColon
   | :: OptionalExpressionallowColon
   | OptionalExpressionnoColon ::

OptionalExpressionα   →
      «empty»
   | CommaExpressionα, allowIn

TypeApplication   →
      .< TypeExpressionList >

MemberExpressionα, β   →
      PrimaryExpressionα, β
   | new MemberExpressionα, β Arguments
   | SuperExpression PropertyOperator
   | MemberExpressionα, β PropertyOperator

SuperExpression   →
      super
   | super ParenExpression

CallExpressionα, β   →
      MemberExpressionα, β Arguments
   | CallExpressionα, β Arguments
   | CallExpressionα, β PropertyOperator

NewExpressionα, β   →
      MemberExpressionα, β
   | new NewExpressionα, β

LeftHandSideExpressionα, β   →
      NewExpressionα, β
   | CallExpressionα, β

PostfixExpressionα, β   →
      LeftHandSideExpressionα, β
   | LeftHandSideExpressionα, β [no line break] ++
   | LeftHandSideExpressionα, β [no line break] --

UnaryExpressionα, β   →
      PostfixExpressionα, β
   | delete PostfixExpressionα, β
   | void UnaryExpressionα, β
   | typeof UnaryExpressionα, β
   | ++ PostfixExpressionα, β
   | -- PostfixExpressionα, β
   | + UnaryExpressionα, β
   | - UnaryExpressionα, β
   | ~ UnaryExpressionα, β
   | ! UnaryExpressionα, β
   | type TypeExpression

MultiplicativeExpressionα, β   →
      UnaryExpressionα, β
   | MultiplicativeExpressionα, β * UnaryExpressionα, β
   | MultiplicativeExpressionα, β / UnaryExpressionα, β
   | MultiplicativeExpressionα, β % UnaryExpressionα, β

AdditiveExpressionα, β   →
      MultiplicativeExpressionα, β
   | AdditiveExpressionα, β + MultiplicativeExpressionα, β
   | AdditiveExpressionα, β - MultiplicativeExpressionα, β

ShiftExpressionα, β   →
      AdditiveExpressionα, β
   | ShiftExpressionα, β << AdditiveExpressionα, β
   | ShiftExpressionα, β >> AdditiveExpressionα, β
   | ShiftExpressionα, β >>> AdditiveExpressionα, β

RelationalExpressionα, β   →
      ShiftExpressionα, β
   | RelationalExpressionα, β < ShiftExpressionα, β
   | RelationalExpressionα, β > ShiftExpressionα, β
   | RelationalExpressionα, β <= ShiftExpressionα, β
   | RelationalExpressionα, β >= ShiftExpressionα, β
   | RelationalExpressionα, β [β == allowIn] in ShiftExpressionα, β
   | RelationalExpressionα, β instanceof ShiftExpressionα, β
   | RelationalExpressionα, β cast TypeExpression
   | RelationalExpressionα, β is TypeExpression
   | RelationalExpressionα, β like TypeExpression

EqualityExpressionα, β   →
      RelationalExpressionα, β
   | EqualityExpressionα, β == RelationalExpressionα, β
   | EqualityExpressionα, β != RelationalExpressionα, β
   | EqualityExpressionα, β === RelationalExpressionα, β
   | EqualityExpressionα, β !== RelationalExpressionα, β

BitwiseAndExpressionα, β   →
      EqualityExpressionα, β
   | BitwiseAndExpressionα, β & EqualityExpressionα, β

BitwiseXorExpressionα, β   →
      BitwiseAndExpressionα, β
   | BitwiseXorExpressionα, β ^ BitwiseAndExpressionα, β

BitwiseOrExpressionα, β   →
      BitwiseXorExpressionα, β
   | BitwiseOrExpressionα, β | BitwiseXorExpressionα, β

LogicalAndExpressionα, β   →
      BitwiseOrExpressionα, β
   | LogicalAndExpressionα, β && BitwiseOrExpressionα, β

LogicalOrExpressionα, β   →
      LogicalAndExpressionα, β
   | LogicalOrExpressionα, β || LogicalAndExpressionα, β

ConditionalExpressionα, β   →
      YieldExpressionα, β
   | LogicalOrExpressionα, β
   | LogicalOrExpressionα, β ? AssignmentExpressionnoColon, β
                  : AssignmentExpressionα, β

NonAssignmentExpressionα, β   →
      YieldExpressionα, β
   | LogicalOrExpressionα, β
   | LogicalOrExpressionα, β ? NonAssignmentExpressionnoColon, β
                  : NonAssignmentExpressionα, β

YieldExpressionα, β   →
      yield
   | yield [no line break] AssignmentExpressionα, β

AssignmentExpressionα, β   →
      ConditionalExpressionα, β
   | Patternα, β, allowExpr = AssignmentExpressionα, β
   | SimplePatternα, β, allowExpr CompoundAssignmentOperator AssignmentExpressionα, β

CompoundAssignmentOperator   →
      *=
   | /=
   | %=
   | +=
   | -=
   | <<=
   | >>=
   | >>>=
   | &=
   | ^=
   | |=
   | &&=
   | ||=

CommaExpressionα, β   →
      AssignmentExpressionα, β
   | CommaExpressionα, β , AssignmentExpressionα, β

 == Patterns == γ = { allowExpr, noExpr }

Patternα, β, γ   →
      SimplePatternα, β, γ
   | ObjectPatternα, β, γ
   | ArrayPatternγ

SimplePatternα, β, noExpr   →
      Identifier

SimplePatternα, β, allowExpr   →
      LeftHandSideExpressionα, β

ObjectPatternγ   →
      { FieldListPatternγ }

FieldListPatternγ   →
      «empty»
   | FieldPatternγ
   | FieldListPatternγ ,
   | FieldListPatternγ , FieldPatternγ

FieldPatternγ   →
      FieldName
   | FieldName : PatternallowColon, allowIn, γ

ArrayPatternγ   →
      [ ElementListPatternγ ]

ElementListPatternγ   →
      «empty»
   | ElementPatternγ
   | ... SimplePatternallowColon, allowIn, γ
   | , ElementListPatternγ
   | ElementPatternγ , ElementListPatternγ

ElementPatternγ   →
      PatternallowColon, allowIn, γ

TypedIdentifier   →
      Identifier
   | Identifier : TypeExpression

TypedPatternβ   →
      PatternnoColon, β, noExpr
   | PatternnoColon, β, noExpr : TypeExpression

LikenedPatternβ   →
      PatternnoColon, β, noExpr like TypeExpression

   | BreakStatement Semicolonω
   | ContinueStatement Semicolonω

   | if ParenExpression SubstatementnoShortIf else Substatementabbrev

   | CaseClausesfull CaseClauseω

DoWhileStatement   →

WhileStatementω   →
      while ParenExpression Substatementω

ForStatementω   →
      for ( ForInitialiser ; OptionalExpressionallowColon ; OptionalExpressionallowColon ) Substatementω
   | for ( ForInBinding in CommaExpressionallowColon, allowIn ) Substatementω
   | for each ( ForInBinding in CommaExpressionallowColon, allowIn ) Substatementω

ForInitialiser   →
      «empty»
   | CommaExpressionallowColon, noIn
   | VariableDefinitionnoIn, τ

ForInBinding   →
      PatternallowColon, noIn, allowExpr
   | VariableDefinitionKindlocal VariableBindingnoIn

ContinueStatement   →
      continue
   | continue [no line break] Identifier

BreakStatement   →
      break
   | break [no line break] Identifier

ReturnStatement   →
      return
   | return [no line break] CommaExpressionallowColon, allowIn

ThrowStatement   →
      throw CommaExpressionallowColon, allowIn

TryStatement   →
      try { Directiveslocal } CatchClauses
   | try { Directiveslocal } CatchClauses finally { Directiveslocal }
   | try { Directiveslocal } finally { Directiveslocal }

CatchClauses   →
      CatchClause

CatchClause   →

SuperStatement   →
